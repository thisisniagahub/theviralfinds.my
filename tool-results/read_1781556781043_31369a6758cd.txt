   171→    setExecutingTool(toolName)
   172→    try {
   173→      const params: Record<string, unknown> = {}
   174→      for (const [k, v] of Object.entries(inputs)) {
   175→        if (v) {
   176→          const num = Number(v)
   177→          params[k] = !isNaN(num) && v !== '' ? num : v
   178→        }
   179→      }
   180→      const res = await fetch('/api/openclaw/mcp-proxy?path=/tools/' + toolName + '/execute', {
   181→        method: 'POST',
   182→        headers: { 'Content-Type': 'application/json' },
   183→        body: JSON.stringify({ jsonrpc: '2.0', id: `exec_${Date.now()}`, params }),
   184→      })
   185→      const data = await res.json()
   186→      if (data.error) {
   187→        toast.error('Execution failed', { description: data.error.message })
   188→      } else {
   189→        const text = data.result?.content?.[0]?.text || JSON.stringify(data.result, null, 2)
   190→        setToolResults(prev => ({ ...prev, [toolName]: text }))
   191→        toast.success(`✅ ${toolName} completed`, { description: 'View results below' })
   192→        queryClient.invalidateQueries({ queryKey: ['mcp-tools'] })
   193→      }
   194→    } catch {
   195→      toast.error('Failed to execute tool', { description: 'MCP Server may be offline' })
   196→    }
   197→    setExecutingTool(null)
   198→  }, [queryClient])

   203→    onSuccess: (data) => {
   204→      if (data.success) {

   206→        toast.success('Content generated!', { description: `${data.wordCount} words, ${data.hashtags?.length || 0} hashtags` })

   208→        toast.error('Content generation failed', { description: data.error })
   209→      }
