    59→export async function POST(request: NextRequest) {
    60→  try {
    61→    const body = await request.json()
    62→    const { product, template, duration, platform, language } = body
    63→
    64→    if (!product || !template || !duration) {
    65→      return NextResponse.json(
    66→        { error: 'Product name, template type, and duration are required' },
    67→        { status: 400 }
    68→      )
    69→    }
    70→
    71→    const templateDesc = TEMPLATE_DESCRIPTIONS[template] || TEMPLATE_DESCRIPTIONS.demo
    72→    const langInstruction = LANGUAGE_INSTRUCTIONS[language || 'english'] || LANGUAGE_INSTRUCTIONS.english
    73→    const platformNote = platform === 'tiktok' ? 'TikTok (vertical 9:16, trending sounds, hashtags important)' :
    74→      platform === 'reels' ? 'Instagram Reels (vertical 9:16, aesthetic, shareable)' :
    75→      platform === 'shorts' ? 'YouTube Shorts (vertical 9:16, SEO-friendly title, thumbnails matter)' :
    76→      'Short-form vertical video platform'
    77→
    78→    const userPrompt = `Create a ${duration} second video script for a Malaysian Shopee affiliate product.
    79→
    80→Product: ${product}
    81→Template Format: ${template} - ${templateDesc}
    82→Platform: ${platformNote}
    83→Language Style: ${langInstruction}
    84→
    85→Remember:
    86→- Include #ad or #promosi disclosure
    87→- Make it feel natural and Malaysian
    88→- Include specific visual/action directions
    89→- Time scenes precisely for ${duration} seconds total
    90→- End with a strong CTA to click the affiliate link
    91→
    92→Return ONLY the JSON object, no markdown code fences.`
    93→
    94→    let scriptResult: Record<string, unknown>
    95→
    96→    try {
    97→      const zai = await ZAI.create()
    98→      const completion = await zai.chat.completions.create({
    99→        messages: [
   100→          { role: 'system', content: SYSTEM_PROMPT },
   101→          { role: 'user', content: userPrompt },
   102→        ],
   103→        thinking: { type: 'disabled' },
   104→      })
   105→
   106→      const content = completion.choices?.[0]?.message?.content || ''
   107→
   108→      // Try to parse JSON from the response
   109→      try {
   110→        // Strip markdown code fences if present
   111→        const jsonStr = content.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim()
   112→        scriptResult = JSON.parse(jsonStr)
   113→      } catch {
   114→        // If JSON parsing fails, create a structured response from the text
   115→        scriptResult = {
   116→          title: `${template} Script - ${product}`,
   117→          totalDuration: `${duration}s`,
   118→          rawScript: content,
   119→          scenes: parseScenesFromText(content, duration),
   120→          hashtags: [`#${template}`, '#shopeeaffiliate', '#ad', '#promosi'],
   121→          tips: ['Speak naturally and authentically', 'Show the product clearly', 'Include your affiliate link in bio'],
   122→        }
   123→      }
   124→    } catch (aiError) {
   125→      console.error('AI script generation error:', aiError)
   126→      scriptResult = generateFallbackScript(product, template, duration, platform, language)
   127→    }
   128→
   129→    return NextResponse.json({
   130→      success: true,
   131→      script: scriptResult,
   132→      meta: {
   133→        product,
   134→        template,
   135→        duration,
   136→        platform,
   137→        language: language || 'english',
   138→        generatedAt: new Date().toISOString(),
   139→      },
   140→    })
   141→  } catch (error) {
   142→    console.error('Script generation error:', error)
   143→    return NextResponse.json(
   144→      { error: 'Failed to generate script' },
   145→      { status: 500 }
   146→    )
   147→  }
   148→}

   150→function parseScenesFromText(text: string, duration: number): Array<Record<string, unknown>> {
   151→  const scenes: Array<Record<string, unknown>> = []
   152→  const lines = text.split('\n').filter((l: string) => l.trim())
   153→
   154→  const sceneCount = duration <= 15 ? 3 : duration <= 30 ? 5 : duration <= 60 ? 6 : 8
   155→  const sceneDuration = Math.floor(duration / sceneCount)
   156→
   157→  const labels = ['Hook', 'Problem', 'Solution', 'Demo', 'Social Proof', 'CTA', 'Bonus', 'Closing']
   158→
   159→  for (let i = 0; i < sceneCount; i++) {
   160→    const start = i * sceneDuration
   161→    const end = i === sceneCount - 1 ? duration : (i + 1) * sceneDuration
   162→    const relevantLines = lines.slice(i * 2, i * 2 + 2).join(' ')
   163→
   164→    scenes.push({
   165→      sceneNumber: i + 1,
   166→      timeRange: `${start}-${end}s`,
   167→      label: labels[i] || `Scene ${i + 1}`,
   168→      visualNotes: `Show ${relevantLines.slice(0, 80) || 'product in action'}`,
   169→      dialogue: relevantLines.slice(0, 150) || 'Voiceover text here',
   170→      actionNotes: 'Perform action as directed',
   171→    })
   172→  }
   173→
   174→  return scenes
   175→}
