   243→// --- QR Code SVG Placeholder ---
   244→function QRCodePlaceholder({ size = 160 }: { size?: number }) {
   245→  // Simple QR code visual pattern
   246→  const cells = 21
   247→  const cellSize = size / cells
   248→
   249→  // Generate a deterministic pattern that looks like a QR code
   250→  const pattern: boolean[][] = []
   251→  for (let r = 0; r < cells; r++) {
   252→    pattern[r] = []
   253→    for (let c = 0; c < cells; c++) {
   254→      // Position detection patterns (corners)
   255→      const isTopLeft = r < 7 && c < 7
   256→      const isTopRight = r < 7 && c >= cells - 7
   257→      const isBottomLeft = r >= cells - 7 && c < 7
   258→
   259→      if (isTopLeft || isTopRight || isBottomLeft) {
   260→        const lr = isTopLeft ? r : isBottomLeft ? r - (cells - 7) : r
   261→        const lc = isTopLeft ? c : isTopRight ? c - (cells - 7) : c
   262→        const isOuterBorder = lr === 0 || lr === 6 || lc === 0 || lc === 6
   263→        const isInnerSquare = lr >= 2 && lr <= 4 && lc >= 2 && lc <= 4
   264→        pattern[r][c] = isOuterBorder || isInnerSquare
   265→      } else {
   266→        // Seeded pseudo-random pattern
   267→        const seed = (r * 31 + c * 17 + 42) % 100
   268→        pattern[r][c] = seed < 40
   269→      }
   270→    }
   271→  }
   272→
   273→  return (
   274→    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
   275→      <rect width={size} height={size} fill="white" rx={4} />
   276→      {pattern.map((row, r) =>
   277→        row.map((cell, c) =>
   278→          cell ? (
   279→            <rect
   280→              key={`${r}-${c}`}
   281→              x={c * cellSize}
   282→              y={r * cellSize}
   283→              width={cellSize}
   284→              height={cellSize}
   285→              fill="black"
   286→            />
   287→          ) : null
   288→        )
   289→      )}
   290→    </svg>
   291→  )
   292→}
