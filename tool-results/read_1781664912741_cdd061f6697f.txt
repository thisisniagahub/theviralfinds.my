   265→// --- QR Code SVG Placeholder ---
   266→function QRCodePlaceholder({ size = 160 }: { size?: number }) {
   267→  const cells = 21
   268→  const cellSize = size / cells
   269→
   270→  const pattern: boolean[][] = []
   271→  for (let r = 0; r < cells; r++) {
   272→    pattern[r] = []
   273→    for (let c = 0; c < cells; c++) {
   274→      const isTopLeft = r < 7 && c < 7
   275→      const isTopRight = r < 7 && c >= cells - 7
   276→      const isBottomLeft = r >= cells - 7 && c < 7
   277→
   278→      if (isTopLeft || isTopRight || isBottomLeft) {
   279→        const lr = isTopLeft ? r : isBottomLeft ? r - (cells - 7) : r
   280→        const lc = isTopLeft ? c : isTopRight ? c - (cells - 7) : c
   281→        const isOuterBorder = lr === 0 || lr === 6 || lc === 0 || lc === 6
   282→        const isInnerSquare = lr >= 2 && lr <= 4 && lc >= 2 && lc <= 4
   283→        pattern[r][c] = isOuterBorder || isInnerSquare
   284→      } else {
   285→        const seed = (r * 31 + c * 17 + 42) % 100
   286→        pattern[r][c] = seed < 40
   287→      }
   288→    }
   289→  }
   290→
   291→  return (
   292→    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
   293→      <rect width={size} height={size} fill="white" rx={4} />
   294→      {pattern.map((row, r) =>
   295→        row.map((cell, c) =>
   296→          cell ? (
   297→            <rect
   298→              key={`${r}-${c}`}
   299→              x={c * cellSize}
   300→              y={r * cellSize}
   301→              width={cellSize}
   302→              height={cellSize}
   303→              fill="black"
   304→            />
   305→          ) : null
   306→        )
   307→      )}
   308→    </svg>
   309→  )
   310→}
