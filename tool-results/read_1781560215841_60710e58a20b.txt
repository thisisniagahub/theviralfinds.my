   244тЖТ   244тЖТfunction QRCodePlaceholder({ size = 160 }: { size?: number }) {
   245тЖТ   245тЖТ  // Simple QR code visual pattern
   246тЖТ   246тЖТ  const cells = 21
   247тЖТ   247тЖТ  const cellSize = size / cells
   248тЖТ   248тЖТ
   249тЖТ   249тЖТ  // Generate a deterministic pattern that looks like a QR code
   250тЖТ   250тЖТ  const pattern: boolean[][] = []
   251тЖТ   251тЖТ  for (let r = 0; r < cells; r++) {
   252тЖТ   252тЖТ    pattern[r] = []
   253тЖТ   253тЖТ    for (let c = 0; c < cells; c++) {
   254тЖТ   254тЖТ      // Position detection patterns (corners)
   255тЖТ   255тЖТ      const isTopLeft = r < 7 && c < 7
   256тЖТ   256тЖТ      const isTopRight = r < 7 && c >= cells - 7
   257тЖТ   257тЖТ      const isBottomLeft = r >= cells - 7 && c < 7
   258тЖТ   258тЖТ
   259тЖТ   259тЖТ      if (isTopLeft || isTopRight || isBottomLeft) {
   260тЖТ   260тЖТ        const lr = isTopLeft ? r : isBottomLeft ? r - (cells - 7) : r
   261тЖТ   261тЖТ        const lc = isTopLeft ? c : isTopRight ? c - (cells - 7) : c
   262тЖТ   262тЖТ        const isOuterBorder = lr === 0 || lr === 6 || lc === 0 || lc === 6
   263тЖТ   263тЖТ        const isInnerSquare = lr >= 2 && lr <= 4 && lc >= 2 && lc <= 4
   264тЖТ   264тЖТ        pattern[r][c] = isOuterBorder || isInnerSquare
   265тЖТ   265тЖТ      } else {
   266тЖТ   266тЖТ        // Seeded pseudo-random pattern
   267тЖТ   267тЖТ        const seed = (r * 31 + c * 17 + 42) % 100
   268тЖТ   268тЖТ        pattern[r][c] = seed < 40
   269тЖТ   269тЖТ      }
   270тЖТ   270тЖТ    }
   271тЖТ   271тЖТ  }
   272тЖТ   272тЖТ
   273тЖТ   273тЖТ  return (
   274тЖТ   274тЖТ    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
   275тЖТ   275тЖТ      <rect width={size} height={size} fill="white" rx={4} />
   276тЖТ   276тЖТ      {pattern.map((row, r) =>
   277тЖТ   277тЖТ        row.map((cell, c) =>
   278тЖТ   278тЖТ          cell ? (
   279тЖТ   279тЖТ            <rect
   280тЖТ   280тЖТ              key={`${r}-${c}`}
   281тЖТ   281тЖТ              x={c * cellSize}
   282тЖТ   282тЖТ              y={r * cellSize}
   283тЖТ   283тЖТ              width={cellSize}
   284тЖТ   284тЖТ              height={cellSize}
   285тЖТ   285тЖТ              fill="black"
   286тЖТ   286тЖТ            />
   287тЖТ   287тЖТ          ) : null
   288тЖТ   288тЖТ        )
   289тЖТ   289тЖТ      )}
   290тЖТ   290тЖТ    </svg>
   291тЖТ   291тЖТ  )
