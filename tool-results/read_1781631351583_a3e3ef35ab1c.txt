   239→function formatDate(date: Date): string {
   240→  const now = new Date()
   241→  const diff = now.getTime() - date.getTime()
   242→  const minutes = Math.floor(diff / 60000)
   243→  const hours = Math.floor(diff / 3600000)
   244→  const days = Math.floor(diff / 86400000)
   245→  if (minutes < 1) return 'Just now'
   246→  if (minutes < 60) return `${minutes}m ago`
   247→  if (hours < 24) return `${hours}h ago`
   248→  return `${days}d ago`
   249→}

   253→}
