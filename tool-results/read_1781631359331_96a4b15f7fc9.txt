   239тЖТ   239тЖТfunction formatDate(date: Date): string {
   240тЖТ   240тЖТ  const now = new Date()
   241тЖТ   241тЖТ  const diff = now.getTime() - date.getTime()
   242тЖТ   242тЖТ  const minutes = Math.floor(diff / 60000)
   243тЖТ   243тЖТ  const hours = Math.floor(diff / 3600000)
   244тЖТ   244тЖТ  const days = Math.floor(diff / 86400000)
   245тЖТ   245тЖТ  if (minutes < 1) return 'Just now'
   246тЖТ   246тЖТ  if (minutes < 60) return `${minutes}m ago`
   247тЖТ   247тЖТ  if (hours < 24) return `${hours}h ago`
   248тЖТ   248тЖТ  return `${days}d ago`
   249тЖТ   249тЖТ}
