    96→export function AgentOfficePage() {
    97→  const [agents, setAgents] = useState<AgentInfo[]>([])
    98→  const [memo, setMemo] = useState<MemoData | null>(null)
    99→  const [memoLoading, setMemoLoading] = useState(true)
   100→  const [activeStatus, setActiveStatus] = useState<AgentStatus | null>(null)
   101→  const [isPaused, setIsPaused] = useState(false)
   102→  const [language, setLanguage] = useState<Language>('en')
   103→  const [showCoords, setShowCoords] = useState(false)
   104→  const [viewMode, setViewMode] = useState<ViewMode>('office')
   105→  const [selectedAgentId, setSelectedAgentId] = useState<string | null>(null)
   106→  const [showMinimap, setShowMinimap] = useState(true)
   107→  const [officeTheme, setOfficeTheme] = useState<OfficeTheme>('night')
   108→  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null)
   109→  const simulationRef = useRef<ReturnType<typeof setInterval> | null>(null)
   110→
   111→  // ===== Fetch initial data =====
   112→  useEffect(() => {
   113→    async function fetchData() {
   114→      try {
   115→        const [agentsRes, memoRes] = await Promise.all([
   116→          fetch('/api/shopee-office/agents'),
   117→          fetch('/api/shopee-office/memo'),
   118→        ])
   119→        if (agentsRes.ok) {
   120→          const data = await agentsRes.json()
   121→          setAgents(data.agents || [])
   122→        }
   123→        if (memoRes.ok) {
   124→          const data = await memoRes.json()
   125→          setMemo(data.memo || null)
   126→        }
   127→      } catch (err) {
   128→        console.error('Failed to fetch shopee-office data:', err)
   129→      } finally {
   130→        setMemoLoading(false)
   131→      }
   132→    }
   133→    fetchData()
   134→  }, [])
   135→
   136→  // ===== Poll agents every 3 seconds =====
   137→  useEffect(() => {
   138→    pollRef.current = setInterval(async () => {
   139→      if (isPaused) return
   140→      try {
   141→        const res = await fetch('/api/shopee-office/agents')
   142→        if (res.ok) {
   143→          const data = await res.json()
   144→          setAgents(data.agents || [])
   145→        }
   146→      } catch {
   147→        // Silent fail for polling
   148→      }
   149→    }, 3000)
   150→
   151→    return () => {
   152→      if (pollRef.current) clearInterval(pollRef.current)
   153→    }
   154→  }, [isPaused])
   155→
   156→  // ===== Simulation: randomize one agent every 5-8s =====
   157→  useEffect(() => {
   158→    if (isPaused) return
   159→
   160→    simulationRef.current = setInterval(async () => {
   161→      if (agents.length === 0) return
   162→      const randomAgent = agents[Math.floor(Math.random() * agents.length)]
   163→      if (!randomAgent || randomAgent.authStatus === 'offline') return
   164→
   165→      const statuses: AgentStatus[] = ['idle', 'writing', 'researching', 'executing', 'syncing', 'error']
   166→      const randomStatus = statuses[Math.floor(Math.random() * statuses.length)]
   167→
   168→      try {
   169→        const res = await fetch(`/api/shopee-office/agents/${randomAgent.agentId}`, {
   170→          method: 'POST',
   171→          headers: { 'Content-Type': 'application/json' },
   172→          body: JSON.stringify({ status: randomStatus }),
   173→        })
   174→        if (res.ok) {
   175→          const data = await res.json()
   176→          setAgents((prev) =>
   177→            prev.map((a) =>
   178→              a.agentId === data.agent.agentId ? { ...a, ...data.agent } : a
   179→            )
   180→          )
   181→        }
   182→      } catch {
   183→        // Silent fail for simulation
   184→      }
   185→    }, 5000 + Math.random() * 3000)
   186→
   187→    return () => {
   188→      if (simulationRef.current) clearInterval(simulationRef.current)
   189→    }
   190→  }, [isPaused, agents])
   191→
   192→  // ===== Handlers =====
   193→  const handleSetAllStatus = useCallback(async (status: AgentStatus) => {
   194→    setActiveStatus(status)
   195→    try {
   196→      const res = await fetch('/api/shopee-office/agents', {
   197→        method: 'POST',
   198→        headers: { 'Content-Type': 'application/json' },
   199→        body: JSON.stringify({ status }),
   200→      })
   201→      if (res.ok) {
   202→        const data = await res.json()
   203→        setAgents(data.agents || [])
   204→      }
   205→    } catch {
   206→      console.error('Failed to set all agents status')
   207→    }
   208→  }, [])
   209→
   210→  const handleSetAgentStatus = useCallback(async (agentId: string, status: string) => {
   211→    try {
   212→      const res = await fetch(`/api/shopee-office/agents/${agentId}`, {
   213→        method: 'POST',
   214→        headers: { 'Content-Type': 'application/json' },
   215→        body: JSON.stringify({ status }),
   216→      })
   217→      if (res.ok) {
   218→        const data = await res.json()
   219→        setAgents((prev) =>
   220→          prev.map((a) =>
   221→            a.agentId === data.agent.agentId ? { ...a, ...data.agent } : a
   222→          )
   223→        )
   224→      }
   225→    } catch {
   226→      console.error('Failed to set agent status')
   227→    }
   228→  }, [])
   229→
   230→  const handleApproveAgent = useCallback(async (agentId: string) => {
   231→    try {
   232→      const res = await fetch(`/api/shopee-office/agents/${agentId}`, {
   233→        method: 'POST',
   234→        headers: { 'Content-Type': 'application/json' },
   235→        body: JSON.stringify({ authStatus: 'approved' }),
   236→      })
   237→      if (res.ok) {
   238→        const data = await res.json()
   239→        setAgents((prev) =>
   240→          prev.map((a) =>
   241→            a.agentId === data.agent.agentId ? { ...a, ...data.agent } : a
   242→          )
   243→        )
   244→      }
   245→    } catch {
   246→      console.error('Failed to approve agent')
   247→    }
   248→  }, [])
   249→
   250→  const handleRejectAgent = useCallback(async (agentId: string) => {
   251→    try {
   252→      const res = await fetch(`/api/shopee-office/agents/${agentId}`, {
   253→        method: 'POST',
   254→        headers: { 'Content-Type': 'application/json' },
   255→        body: JSON.stringify({ authStatus: 'offline' }),
   256→      })
   257→      if (res.ok) {
   258→        const data = await res.json()
   259→        setAgents((prev) =>
   260→          prev.map((a) =>
   261→            a.agentId === data.agent.agentId ? { ...a, ...data.agent } : a
   262→          )
   263→        )
   264→      }
   265→    } catch {
   266→      console.error('Failed to reject agent')
   267→    }
   268→  }, [])
   269→
   270→  // ===== Profile back handler =====
   271→  const handleProfileBack = useCallback(() => {
   272→    setSelectedAgentId(null)
   273→    setViewMode('grid')
   274→  }, [])
   275→
   276→  // ===== Keyboard shortcuts =====
   277→  useEffect(() => {
   278→    function handleKeyDown(e: KeyboardEvent) {
   279→      if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return
   280→
   281→      const statusMap: Record<string, AgentStatus> = {
   282→        '1': 'idle',
   283→        '2': 'writing',
   284→        '3': 'researching',
   285→        '4': 'executing',
   286→        '5': 'syncing',
   287→        '6': 'error',
   288→      }
   289→
   290→      if (statusMap[e.key]) {
   291→        e.preventDefault()
   292→        handleSetAllStatus(statusMap[e.key])
   293→      }
   294→
   295→      if (e.key === ' ') {
   296→        e.preventDefault()
   297→        setIsPaused((prev) => !prev)
   298→      }
   299→
   300→      if (e.key === 'm' || e.key === 'M') {
   301→        setShowMinimap((prev) => !prev)
   302→      }
   303→
   304→      // Tab switching shortcuts
   305→      if (e.key === 'q') setViewMode('office')
   306→      if (e.key === 'w') setViewMode('grid')
   307→      if (e.key === 'e') setViewMode('pipeline')
   308→      if (e.key === 'r') setViewMode('isometric')
   309→      if (e.key === 't') setViewMode('chat')
   310→    }
   311→
   312→    window.addEventListener('keydown', handleKeyDown)
   313→    return () => window.removeEventListener('keydown', handleKeyDown)
   314→  }, [handleSetAllStatus])
   315→
   316→  // ===== Map agents to Phaser format =====
   317→  const phaserAgents: AgentData[] = agents
   318→    .filter((a) => a.authStatus !== 'offline')
   319→    .map(mapApiAgentToPhaser)
   320→
   321→  const gridAgents: AgentGridAgent[] = agents.map(mapApiAgentToGrid)
   322→  const profileAgent: ProfileAgent | null = selectedAgentId
   323→    ? mapApiAgentToProfile(agents.find((a) => a.agentId === selectedAgentId) || agents[0])
   324→    : null
   325→
   326→  // ===== Status text for overlay =====
   327→  const activeAgents = agents.filter((a) => a.status !== 'idle' && a.status !== 'error')
   328→  const errorAgents = agents.filter((a) => a.status === 'error')
   329→  const onlineAgents = agents.filter((a) => a.authStatus !== 'offline')
   330→  const totalTasksCompleted = agents.reduce((sum, a) => sum + a.tasksCompleted, 0)
   331→  const avgTasks = agents.length > 0 ? Math.round(totalTasksCompleted / agents.length) : 0
   332→  const productivityRate = onlineAgents.length > 0 ? Math.round((activeAgents.length / onlineAgents.length) * 100) : 0
   333→  let statusText = `${activeAgents.length}/${onlineAgents.length} active`
   334→  if (errorAgents.length > 0) {
   335→    statusText += ` | ${errorAgents.length} error${errorAgents.length > 1 ? 's' : ''}`
   336→  }
   337→  if (isPaused) {
   338→    statusText = `⏸ PAUSED — ${statusText}`
   339→  }
   340→
   341→  // ===== View Labels =====
   342→  const viewLabels: Record<Language, Record<ViewMode, string>> = {
   343→    en: { office: 'Office', isometric: 'Isometric', grid: 'Agents', profile: 'Profile', pipeline: 'Pipeline', chat: 'Chat', performance: 'Stats' },
   344→    cn: { office: '办公室', isometric: '等距视图', grid: '代理', profile: '档案', pipeline: '流水线', chat: '聊天', performance: '统计' },
   345→    jp: { office: 'オフィス', isometric: '等角図', grid: 'エージェント', profile: 'プロファイル', pipeline: 'パイプライン', chat: 'チャット', performance: '統計' },
   346→  }
   347→
   348→  return (
   349→    <div className={`shopee-office-container office-theme-${officeTheme === 'auto' ? 'night' : officeTheme}`}>
   350→      {/* ===== Header Bar ===== */}
   351→      <motion.div
   352→        className="flex items-center justify-between px-4 max-w-[1280px] mx-auto flex-wrap gap-2"
   353→        initial={{ opacity: 0 }}
   354→        animate={{ opacity: 1 }}
   355→        transition={{ duration: 0.3 }}
   356→      >
   357→        <div className="flex items-center gap-3">
   358→          <span className="text-xl font-bold text-[#EE4D2D] font-mono tracking-wide">
   359→            🛒 Shopee Office
   360→          </span>
   361→          <div className={`shopee-pause-indicator ${isPaused ? 'paused' : 'running'}`}>
   362→            <span className="w-2 h-2 rounded-full" style={{
   363→              background: isPaused ? '#eab308' : '#22c55e',
   364→              animation: isPaused ? 'none' : 'dotPulse 1.5s ease-in-out infinite',
   365→            }} />
   366→            <span>{isPaused ? 'PAUSED' : 'LIVE'}</span>
   367→          </div>
   368→          {/* Version badge */}
   369→          <span className="hidden sm:inline-flex px-1.5 py-0.5 rounded text-[8px] font-bold bg-[#EE4D2D15] text-[#EE4D2D] border border-[#EE4D2D33] font-mono">
   370→            v7.0
   371→          </span>
   372→        </div>
   373→
   374→        {/* View Mode Tabs */}
   375→        <div className="flex items-center gap-1 bg-[#0d1020] border border-[#2a2d3e] rounded-md p-1 overflow-x-auto" style={{ scrollbarWidth: 'none' }}>
   376→          <ViewTab
   377→            label={viewLabels[language].office}
   378→            icon="🏢"
   379→            isActive={viewMode === 'office'}
   380→            onClick={() => setViewMode('office')}
   381→          />
   382→          <ViewTab
   383→            label={viewLabels[language].isometric}
   384→            icon="🏗️"
   385→            isActive={viewMode === 'isometric'}
   386→            onClick={() => setViewMode('isometric')}

   388→          <ViewTab
   389→            label={viewLabels[language].grid}
   390→            icon="👥"
   391→            isActive={viewMode === 'grid'}
   392→            onClick={() => setViewMode('grid')}
   393→          />
   394→          <ViewTab
   395→            label={viewLabels[language].pipeline}
   396→            icon="🔗"
   397→            isActive={viewMode === 'pipeline'}
   398→            onClick={() => setViewMode('pipeline')}
   399→          />
   400→          <ViewTab
   401→            label={viewLabels[language].chat}
   402→            icon="💬"
   403→            isActive={viewMode === 'chat'}
   404→            onClick={() => setViewMode('chat')}
   405→          />
   406→          <ViewTab
   407→            label={viewLabels[language].performance}
   408→            icon="📊"
   409→            isActive={viewMode === 'performance'}

   411→          />
   412→          {viewMode === 'profile' && (
   413→            <ViewTab
   414→              label={viewLabels[language].profile}
   415→              icon="📋"
   416→              isActive={viewMode === 'profile'}

   421→
   422→        <div className="flex items-center gap-2">
   423→          {/* Quick stats in header */}
   424→          <div className="hidden sm:flex items-center gap-4 mr-2">
   425→            <span className="text-[10px] font-mono text-gray-400">
   426→              Agents: <span className="text-white font-bold">{onlineAgents.length}</span>
   427→            </span>
   428→            <span className="text-[10px] font-mono text-gray-400">
   429→              Active: <span className="text-green-400 font-bold">{activeAgents.length}</span>
   430→            </span>
   431→          </div>

   472→              {/* Status overlay */}

   484→              {/* Minimap toggle button */}

   499→            {/* ===== Productivity Stats Bar ===== */}
